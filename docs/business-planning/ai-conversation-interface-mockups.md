### Real-time Data Flow
```typescript
// WebSocket connection for live updates
const useConversationSocket = (sessionId: string) => {
  const [messages, setMessages] = useState<Message[]>([]);
  const [consensus, setConsensus] = useState<ConsensusState[]>([]);
  const [simulationResults, setSimulationResults] = useState<SimulationResult>();

  useEffect(() => {
    const ws = new WebSocket(`/ws/conversation/${sessionId}`);
    
    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      switch (data.type) {
        case 'message':
          setMessages(prev => [...prev, data.payload]);
          break;
        case 'consensus_update':
          setConsensus(data.payload);
          break;
        case 'simulation_result':
          setSimulationResults(data.payload);
          break;
      }
    };

    return () => ws.close();
  }, [sessionId]);
};
```
